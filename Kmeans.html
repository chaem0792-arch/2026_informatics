function drawStar(cx,cy,r,n,color){
ctx.save(); ctx.beginPath();
for(let i=0;i<n*2;i++){
const angle=(i*Math.PI/n)-Math.PI/2;
const radius=i%2===0?r:r*0.45;
const x=cx+radius*Math.cos(angle), y=cy+radius*Math.sin(angle);
i===0?ctx.moveTo(x,y):ctx.lineTo(x,y);
}
ctx.closePath(); ctx.fillStyle=color; ctx.fill();
ctx.strokeStyle='#000'; ctx.lineWidth=1; ctx.stroke();
ctx.restore();
}
 
const hexCache={};
function parseHex(hex){
if(hexCache[hex]) return hexCache[hex];
const r=parseInt(hex.slice(1,3),16), g=parseInt(hex.slice(3,5),16), b=parseInt(hex.slice(5,7),16);
hexCache[hex]=[r,g,b]; return [r,g,b];
}

}

function assignClusters(){

centroids.forEach((c,i)=>{ const d=(pt.x-c.x)**2+(pt.y-c.y)**2; if(d<minD){minD=d;best=i;} });

function moveCentroids(){
const newC=centroids.map((_,i)=>{
const pts=dataset.filter((_,j)=>assignments[j]===i);
if(!pts.length) return centroids[i];
const mx=pts.reduce((s,p)=>s+p.x,0)/pts.length;
const my=pts.reduce((s,p)=>s+p.y,0)/pts.length;
return {x:mx,y:my};
});
const moved=newC.some((c,i)=>(c.x-centroids[i].x)**2+(c.y-centroids[i].y)**2>1e-10);
centroids=newC;
return moved;
}

function moveCentroids(){
  const newC=centroids.map((_,i)=>{
    const pts=dataset.filter((_,j)=>assignments[j]===i);
    if(!pts.length) return centroids[i];
    const mx=pts.reduce((s,p)=>s+p.x,0)/pts.length;
    const my=pts.reduce((s,p)=>s+p.y,0)/pts.length;
    return {x:mx,y:my};
  });
  const moved=newC.some((c,i)=>(c.x-centroids[i].x)**2+(c.y-centroids[i].y)**2>1e-10);
  centroids=newC;
  return moved;
}

canvas.addEventListener('mousemove',e=>{
  if(isRunning||isDone||!dataset.length) return;
  const rect=canvas.getBoundingClientRect();
  const mx=(e.clientX-rect.left)*(canvas.width/rect.width);
  const my=(e.clientY-rect.top)*(canvas.height/rect.height);
  let minD=Infinity, nearestIdx=-1;
  dataset.forEach((pt,i)=>{
    const [px,py]=toPixel(pt.x,pt.y);
    const d=Math.sqrt((px-mx)**2+(py-my)**2);
    if(d<28&&d<minD){minD=d;nearestIdx=i;}
  });
  if(nearestIdx!==hoveredPointIdx){
    hoveredPointIdx=nearestIdx;
    canvas.style.cursor=(nearestIdx>=0&&centroids.length<k)?'pointer':'crosshair';
    drawAll();
  }
});
canvas.addEventListener('mouseleave',()=>{ hoveredPointIdx=-1; drawAll(); });
canvas.addEventListener('click',e=>{
  if(isRunning||isDone) return;
  if(centroids.length>=k){ showToast(`이미 ${k}개의 중심점을 선택했습니다`); return; }
  const rect=canvas.getBoundingClientRect();
  const mx=(e.clientX-rect.left)*(canvas.width/rect.width);
  const my=(e.clientY-rect.top)*(canvas.height/rect.height);
  let minD=Infinity,nearestIdx=-1;
  dataset.forEach((pt,i)=>{
    const [px,py]=toPixel(pt.x,pt.y);
    const d=Math.sqrt((px-mx)**2+(py-my)**2);
    if(d<36&&d<minD){minD=d;nearestIdx=i;}
  });
  if(nearestIdx<0){ showToast('데이터 포인트 근처를 클릭하세요'); return; }
  if(centroids.some(c=>c._srcIdx===nearestIdx)){ showToast('이미 선택된 포인트입니다'); return; }
 
  const pt=dataset[nearestIdx];
  centroids.push({x:pt.x,y:pt.y,_srcIdx:nearestIdx});
  updateCentroidList();
  if(centroids.length===k){
    assignClusters();
    setStatus('waiting','중심점 선택 완료! 실행 버튼을 누르세요.');

    showToast(`🎯 ${k}개 중심점 선택 완료!`);
  } else {
    setStatus('',`${centroids.length}/${k}개 선택됨`);
  }
  drawAll();
});
 
function updateCentroidList(){
  const list=document.getElementById('centroidList');
  list.innerHTML='';
  const xObj=ATTRIBUTES.find(a=>a.id===X_ATTR);
  const yObj=ATTRIBUTES.find(a=>a.id===Y_ATTR);
  for(let i=0;i<k;i++){
    const c=centroids[i];
    let coordText='미선택';
    if(c){
      const xVal=(xObj.min+c.x*(xObj.max-xObj.min)).toFixed(xObj.max<=5?1:0);
      const yVal=(yObj.min+c.y*(yObj.max-yObj.min)).toFixed(yObj.max<=5?1:0);
      coordText=`${xVal} / ${yVal}`;
      if(c._srcIdx!==undefined){
        const u=USERS[dataset[c._srcIdx].userId];
        coordText=u.name+' ('+coordText+')';
      }
    }
    const div=document.createElement('div');
    div.className='centroid-item'+(c?' set':'');
    div.innerHTML=`
      <div class="centroid-dot" style="background:${COLORS[i]}"></div>
      <span style="flex:1">C${i+1}</span>
      <span class="centroid-coords" style="font-size:10px">${coordText}</span>
    `;
    list.appendChild(div);
  }
}

function getClusterTags(avgs){
  const tags=[];
  const a=ATTRIBUTES.find(a=>a.id==='action');
  const e=ATTRIBUTES.find(a=>a.id==='emotional');
  const f=ATTRIBUTES.find(a=>a.id==='frequency');
  const r=ATTRIBUTES.find(a=>a.id==='rating');
 
  const aNorm=(avgs.action-a.min)/(a.max-a.min);
  const eNorm=(avgs.emotional-e.min)/(e.max-e.min);
  const fNorm=(avgs.frequency-f.min)/(f.max-f.min);
  const rNorm=(avgs.rating-r.min)/(r.max-r.min);
 
  if(aNorm>0.6) tags.push('🎬 액션 선호');
  else if(aNorm<0.35) tags.push('🎭 액션 비선호');
  if(eNorm>0.6) tags.push('💛 감성파');
  else if(eNorm<0.35) tags.push('🧊 감성 낮음');
  if(fNorm>0.55) tags.push('📺 헤비 시청자');
  else if(fNorm<0.3) tags.push('🌱 라이트 유저');

  return tags;
}
